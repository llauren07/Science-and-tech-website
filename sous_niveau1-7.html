Corrigé de l'exercice "L'espion est démasqué !":


nbPersonnes=int(input())
for i in range(nbPersonnes):
    taille=int(input())
    age=int(input())
    poids=int(input())
    cheval=int(input())
    brun=int(input())
    nbCriteres=0
    if taille>=178 and taille<=182:
        nbCriteres+=1
    if age>=34:
        nbCriteres+=1
    if poids<70:
        nbCriteres+=1
    if cheval==0:
        nbCriteres+=1
    if brun==1:
        nbCriteres+=1
    if nbCriteres==5:
        print("Très probable")
    elif nbCriteres==3 or nbCriteres==4:
        print("Probable")
    elif nbCriteres==0:
        print("Impossible")
    else:
        print("Peu probable")

Zones de couleurs

Corrigé de l'exercice "Zones de couleurs":


nbJetons=int(input())
for i in range(nbJetons):
   x=int(input())
   y=int(input())
   if x>90 or y>70 or x<0 or y < 0:
      print("En dehors de la feuille")
   elif (x>=15 and x<=45 and y>=60 and y<=70 )or (x>=60 and x<=85 and y>=60 and y<=70):
      print("Dans une zone rouge")
   elif (x>=10 and x<=25 and y>=10 and y<=55) or (x>=25 and x<=85 and y>=10 and y<=20) or (x>=50 and x<=85 and y>=20 and y<=55) or (x>=25 and x<=50 and y>=45 and y<=55):
      print("Dans une zone bleue")
   else:
      print("Dans une zone jaune")
